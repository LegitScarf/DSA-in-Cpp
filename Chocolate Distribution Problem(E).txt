class Solution {
  public:
    int findMinDiff(vector<int>& a, int m) {
        sort(a.begin(), a.end());
        int i = 0;
        int j = m-1;
        int n = a.size();
        int mini = INT_MAX;
        
        while(j<n){
            int diff = a[j] - a[i];
            mini = min(mini, diff);
            i++;
            j++;
        }
        return mini;
        
    }
};
