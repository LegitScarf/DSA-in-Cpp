class Solution {
  public:
    int minimumDays(int S, int N, int M) {
        if (M > N) return -1;
        int sunday = S/7;
        int buyingDays = S - sunday;
        //int ans = 0;
        if (S > 6 && (M * 7) > (N * 6)) return -1;
        int totalFood = S*M;
        // if(totalFood%N == 0) ans = totalFood/N;
        // else ans = totalFood/N + 1;
        int ans = (totalFood + N - 1) / N;
        if(ans <= buyingDays) return ans;
        else return -1;
    }
};
